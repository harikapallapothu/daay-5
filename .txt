function countPositive(arr)
{
    let count = 0, sum =0;
    for(i = 0; i<arr.length; i++)
    {
        if(arr[i]>0)
        {
            count+=1;
        }
        if(arr[i]<0)
        {
            sum = sum + (-1*arr[i]);
        }
    }
    return {
        count, sum};
}
